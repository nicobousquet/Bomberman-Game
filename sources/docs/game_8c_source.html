24static struct map *read_new_map(char *filename) {
25 assert(filename);
26 /* opening map file */
27 FILE *fp = fopen(filename, "rb");
28 if (!fp) {
29 perror("fopen");
30 exit(EXIT_FAILURE);
31 }
32
33 /* getting size of file */
34 fseek(fp, 0, SEEK_END);
35 long int size = ftell(fp);
36 fseek(fp, 0, SEEK_SET);
37 char *grid = malloc(size);
38 if (!grid) {
39 perror("malloc");
40 exit(EXIT_FAILURE);
41 }
42
43 /* reading file and setting map->grid */
44 size_t numread = fread(grid, sizeof(char), size, fp);
45 if ((long int) numread != size) {
46 perror("fread");
47 exit(EXIT_FAILURE);
48 }
49 fclose(fp);
50
51 int width = strtol(grid, &grid, 10);
52 int height = strtol(grid + 1, &grid, 10);
53 struct map *map = map_new(width, height);
54 for (int i = 0; i < width * height; i++) {
55 int ret = strtol(grid, &grid, 10);
56 map_get_grid(map)[i] = (unsigned char) ret;
57 }
58
59 return map;
60}

355static short input_keyboard(struct game *game) {
356 assert(game);
357 SDL_Event event = {0};
358 int move = 0;
359 struct player *player = game_get_player(game);
360 struct map *map = game_get_current_map(game);
361 while (SDL_PollEvent(&event)) {
362 switch (event.type) {
363 /* quit game with no backup */
364 case SDL_QUIT:
365 return 1;
366 case SDL_KEYDOWN: {
367 switch (event.key.keysym.sym) {
368 /* Ctrl + S ==> quit game with backup */
369 case SDLK_s:
370 if (event.key.keysym.mod & KMOD_CTRL) {
371 backup_game(game);
372 return 1;
373 }
374 break;
375 /* going north */
376 case SDLK_UP:
377 player_set_current_way(player, NORTH);
378 if (map_move_player(map, player)) {
379 move = 1;
380 }
381 break;
382 /* going south */
383 case SDLK_DOWN:
384 player_set_current_way(player, SOUTH);
385 if (map_move_player(map, player)) {
386 move = 1;
387 }
388 break;
389 /* going east */
390 case SDLK_RIGHT:
391 player_set_current_way(player, EAST);
392 if (map_move_player(map, player)) {
393 move = 1;
394 }
395 break;
396 /* going west */
397 case SDLK_LEFT:
398 player_set_current_way(player, WEST);
399 if (map_move_player(map, player)) {
400 move = 1;
401 }
402 break;
403 /* set a bomb */
404 case SDLK_SPACE:
405 map_set_bomb(map, player);
406 break;
407 /* pause game */
408 case SDLK_p:
409 pause_game(&event);
410 break;
411 case SDLK_RETURN: {
412 /* open door */
413 int x_next_player = direction_get_x(player_get_x(player), player_get_direction(player), 1);
414 int y_next_player = direction_get_y(player_get_y(player), player_get_direction(player), 1);
415
416 if (map_is_inside(map, x_next_player, y_next_player)) {
417 uint8_t type = map_get_cell_value(map, x_next_player, y_next_player);
418 if ((type & 0xf1) == (CELL_DOOR | CLOSED)) {
419 map_set_cell_value(map, x_next_player, y_next_player, type & 0xfe);
420 player_dec_num_keys(player);
421 }
422 }
423 break;
424 }
425 default:
426 break;
427 }
428
429 if (move) {
430 uint8_t cell = map_get_cell_value(map, player_get_x(player), player_get_y(player));
431 if ((cell & 0xf0) == CELL_DOOR) {
432 /* level of next level */
433 int level = (cell & 0x0e) / 2;
434 change_current_level(game, level);
435 }
436 }
437 break;
438 }
439 default:
440 break;
441 }
442 }
443 return 0;
444}
